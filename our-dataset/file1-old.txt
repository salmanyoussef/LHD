public class StatsExample {
    public static void main(String[] args) {
        int[] values = new int[] {10, 20, 30, 40};
        int sum = 0;
        for (int v : values) {
            sum += v;
        }
        double avg = sum / (double) values.length;
        System.out.println("Average: " + avg);
    }
}
