def is_prime(x):
    if x < 2: 
        return False
    for d in range(2, int(x**0.5) + 1):
        if x % d == 0:
            return False
    return True

def primes_up_to(n):
    result = []
    for i in range(2, n + 1):
        if is_prime(i):
            result.append(i)
    return result

print(primes_up_to(20))
