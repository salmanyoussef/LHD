def add(a, b):
    return a + b
def subtract(a, b):
    return a - b
def multiply(a, b):
    return a * b
def divide(a, b):
    if b == 0:
        raise ValueError("Division by zero is not allowed")
    return a / b
def average(numbers):
    if not numbers:
        raise ValueError("numbers cannot be empty")
    total = 0
    for n in numbers:
        total += n
    return total / len(numbers)
