def count_lines(path):
    count = 0
    with open(path, "r", encoding="utf-8") as f:
        for _ in f:
            count += 1
    return count
def main():
    filename = input("Enter a file name: ")
    total = count_lines(filename)
    print("Total lines:", total)
if __name__ == "__main__":
    main()
