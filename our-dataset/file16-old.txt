public class EvenSum {
    public static void main(String[] args) {
        int[] data = new int[] {3, 8, 12, 5, 20};
        int sum = 0;
        for (int v : data) {
            if (v % 2 == 0) {
                sum += v;
            }
        }
        System.out.println("Sum of even numbers: " + sum);
    }
}
