public class Loader {
    public static String load(String path) {
        java.io.FileInputStream fs = null;
        try {
            fs = new java.io.FileInputStream(path);
            byte[] data = fs.readAllBytes();
            return new String(data);
        } catch (Exception e) {
            return "";
        } finally {
            if (fs != null) {
                try { fs.close(); } catch (Exception ignore) {}
            }
        }
    }

    public static void main(String[] args) {
        String result = load("input.txt");
        System.out.println("Loaded:");
        System.out.println(result);
    }
}
