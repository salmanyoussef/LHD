def sort_dict_by_value(d):
    arr = list(d.items())
    for i in range(len(arr)):
        for j in range(i + 1, len(arr)):
            if arr[j][1] < arr[i][1]:
                arr[i], arr[j] = arr[j], arr[i]
    return arr

items = {"apple": 4, "banana": 2, "pear": 9}
sorted_items = sort_dict_by_value(items)
print(sorted_items)
